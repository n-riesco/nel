/** @module log
 *
 * @description Module for logging. Uses module `debug` if installed, otherwise
 * logs to stderr.
 *
 */

module.exports = newLogger;

function newLogger(tag) {
    if (process.env.DEBUG) global.DEBUG = true;

    if (!global.DEBUG) return function dontLog() {};

    var log;
    try {
        log = require("debug")(tag);

    } catch (err) {
        var console = require("console");

        tag += " ";
        log = function log() {
            process.stderr.write(tag);
            console.error.apply(console, arguments);
        };
    }

    return log;
}
